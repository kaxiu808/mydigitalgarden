```js
// 函数声明
function foo(){ 
	let a = "apple";
	console.log(a);
}
// 函数调用
foo()    // 输出：apple
console.log(a)  // 输出：a is not defined  
```

function foo(){  
    console.log("foo功能")

    console.log("bar功能")  
}

    function dec() {

}
